/*WHAT IS THE COUNTRY THAT BUY THE MOST JAZZ SONGS?*//* Query 4-query used for the fourth insight*/Select Customer.Country, count(InvoiceLine.quantity) as qty from customerJOIN Invoice on Customer.CustomerId=Invoice.CustomerIdJOIN InvoiceLine on invoice.InvoiceId=InvoiceLine.InvoiceIdJOIN Track on InvoiceLine.trackId= Track.TrackIdJOIN Genre on Track.GenreId=Genre.GenreIdWHERE Genre.Name ="Jazz"Group by Customer.country
