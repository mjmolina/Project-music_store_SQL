/*WHICH ARE THE TOP 5 GENRES THAT SELL THE MOST?*//*Figure 1. Top 5 Genres.*//* Query 1-query used for the first insight*/SELECT Genre.GenreId, Genre.Name, sum(InvoiceLine.UnitPrice *InvoiceLine.Quantity) as totalFROM Genre JOIN Track on Genre.genreId= Track.GenreIdJOIN InvoiceLine on Track.TrackId = InvoiceLine.TrackIdGROUP BY Genre.GenreIdORDER BY total desc Limit 5;
